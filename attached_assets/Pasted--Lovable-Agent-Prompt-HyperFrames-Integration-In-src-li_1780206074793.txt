  Fragment shader pattern (domain warp FBM):
    float hash(vec2 p) { return fract(sin(dot(p, vec2(127.1, 311.7))) * 43758.5453); }
    float noise(vec2 p) {
      vec2 i = floor(p), f = fract(p);
      f = f * f * (3.0 - 2.0 * f);
      return mix(mix(hash(i), hash(i+vec2(1,0)), f.x), mix(hash(i+vec2(0,1)), hash(i+vec2(1,1)), f.x), f.y);
    }
    float fbm(vec2 p) {
      float v=0., a=0.5; mat2 R=mat2(0.8,0.6,-0.6,0.8);
      for(int i=0;i<5;i++){v+=a*noise(p);p=R*p*2.02;a*=0.5;}
      return v;
    }
    // Cosine palette for color
    vec3 palette(float t) {
      return vec3(0.5)+vec3(0.5)*cos(6.28*(vec3(1)*t+vec3(0,0.33,0.67)));
    }
    // Domain warp main:
    vec2 q = vec2(fbm(uv*3.+t*.3), fbm(uv*3.+vec2(5.2,1.3)+t*.2));
    vec2 r = vec2(fbm(uv*3.+q*4.+vec2(1.7,9.2)+t*.15), fbm(uv*3.+q*4.+vec2(8.3,2.8)+t*.1));
