  }

── 7C. PARTICLE BURST (on element appear) ──
  function particleBurst(x, y, color = '#a3e635', count = 16) {
    for (let i = 0; i < count; i++) {
      const p = document.createElement('div');
      p.style.cssText = `position:fixed;width:${gsap.utils.random(3,7)}px;
        height:${gsap.utils.random(3,7)}px;border-radius:50%;
        background:${color};left:${x}px;top:${y}px;pointer-events:none;z-index:7777`;
      document.body.appendChild(p);
      const angle = (i / count) * Math.PI * 2;
      const dist  = gsap.utils.random(40, 120);
      gsap.to(p, {
        x: Math.cos(angle) * dist,
        y: Math.sin(angle) * dist,
        opacity: 0, scale: 0,
        duration: gsap.utils.random(0.4, 0.8),
        ease: 'power2.out',
        onComplete: () => p.remove()
      });
    }
  }
